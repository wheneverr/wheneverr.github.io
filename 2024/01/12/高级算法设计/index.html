23/11/01 算法实验

【问题描述】对n个数进行k次操作，每次操作选择一个数将最低位的1变成0，求k次操作后得到的数组的最小和。

【输入】n， k，数组a[n]

【输出】最小和

#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int getsum(vector<int>a) {
    int sum = 0;
    for (int i = 0; i < a.size(); i++) {
        sum += a[i];
    }
    return sum;
}

// 对该数进行t次操作
int opration(int num, int t) {
    if(num == 0) return num;
    else{
        while (t > 0 && num != 0)
            {
                num = num - (num & (-num));
                --t;
            }

        return num;
    }
}

vector<int> getpopca(vector<int>a, int n){
    vector<int> popca(n);
    for(int i = 0; i < n; i++){
        int t = 0;
        int num = a[i];
        while (num != 0)
            {
                num = num - (num & (-num));
                ++t;
            }
        popca[i] = t;
    }
    return popca;
}

int dpsolution(vector<int>&a, int n, int k) {
    int dp[100][100];
    vector<int> popca = getpopca(a, n);
    dp[0][0] = a[0];
    for(int i = 1; i < n; i++){
        dp[0][i] = dp[0][i-1] + a[i];
    }
    for(int i = 1; i <= k; i++){
        dp[i][0] = opration(dp[i-1][0], 1);
    }
    for(int i = 1; i<= k; i++){
        for(int j = 1; j< n ; j++){
            int minsum = dp[i][j-1] + a[j];
            for(int k = 1; k <= popca[j]; k++){
                if(i - k < 0) break;
                minsum = min(minsum, dp[i-k][j-1] + opration(a[j], k));
            }
            dp[i][j] = minsum;
        }
    }
    return dp[k][n-1];
}

【问题描述】假设关系R(a,b), S(b,c)，其中R 包含｜R｜个元组，S包含｜S｜ 个元组，R中属性b的取值个数是V(R,b),S中属性取值个数是V(S,b)。关系R和S的自然连接运算结果中元组的个数是｜R｜*｜S｜/max(V(R,b),V(S,b)).
现有N 个关系R1，R2，。。。。Rn,他们每个均包含两个属性，Ri和Ri+1有一个相同的属性，Rn和R1有一个相同的属性，，请设计他们连接的顺序，使得连接形成的中间结果最少，即连接的代价最少。
【输入】
第一行，一个整数N代表关系的个数。
下N行，每一行三个整数，空格隔开，代表关系Ri(a,b)中元组的个数，以及V(Ri,a)，V(Ri,b)中值的个数。
【输出】
最小连接代价的和。

#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int sumcost = 0;

int calculateCost(int R, int S,int Vr, int Vs) {
    return R * S / max(Vr, Vs);
}

void getCostArr(vector<vector<int>> R, int N, int s[][100]){
    for(int len = 1; len < N; len++){
        for( int i = 0; i < N * 2 - len; i++){
            if(len == 1)
                s[i][i+len] = calculateCost(R[i][0], R[i+len][0], R[i][2], R[i+len][1]);
            else if(len == N-1)
                s[i][i+len] = calculateCost(s[i][i+len-1], R[i+len][0], R[i+len-1][2], R[i+len][1]) / max(R[i][1],R[i+len][2]);
            else{
                s[i][i+len] = calculateCost(s[i][i+len-1], R[i+len][0], R[i+len-1][2], R[i+len][1]);
            }
        }
    }
}

int getMinCost(vector<vector<int>> R,int N, int s[][100]) {
    vector<vector<int>> dp(N, vector<int>(N * 2)); // dp[i][j]是i到j合并的最小代价
    for(int len = 1; len < N; len++){
        for( int i = 0; i < N * 2 - len; i++){
            if(len == 1){
                dp[i][i+1] = s[i][i+1];
            }
            else if(len ==2){
                dp[i][i+2] = min(dp[i][i+1] + calculateCost(s[i][i+1], R[i+2][0], R[i+1][2], R[i+2][1]),
                    dp[i+1][i+2] + calculateCost(R[i][0], s[i+1][i+2], R[i][2], R[i+1][1]));
            }
            else if(len == N-1){
                int mincost = min(s[i][i+len] + dp[i+1][i+len], s[i][i+len] + dp[i][i+len-1]);
                for(int k = i+1; k<i+len-1;k++){
                    mincost = min(s[i][i+len] + dp[i][k] + dp[k+1][i+len], mincost);
                }
                dp[i][i+len] = mincost;
            }
            else{
                int mincost = min(calculateCost(R[i][0], s[i+1][i+len], R[i][2], R[i+1][1]) + dp[i+1][i+len],
                calculateCost(s[i][i+len-1],R[i+len][0], R[i+len-1][2], R[i+len][1]) + dp[i][i+len-1]);
                for( int k = i+1; k<i+len-1;k++){
                    mincost = min(calculateCost(s[i][k], s[k+1][i+len], R[k][2], R[k+1][1]) + dp[i][k] + dp[k+1][i+len], mincost);
                }
                dp[i][i+len] = mincost;
            }
        }
    }
    int sum = dp[0][N-1];
    for(int i = 0; i< N; i++){
        sum = min(dp[i][i + N -1], sum);
    }

    for(int i = 0; i< N*2; i++){
        for(int j = 0; j<N*2;j++){
            cout<<dp[i][j]<<" ";
        }
        cout<<endl;
    }
    return sum;
}

1、套汇问题

【问题描述】
设已知n种货币C1，C2，，Cn以及对应的汇率矩阵R,其中R[][]是第i种单位货币
可以购买第j种货币的单位数。若存在R[i1][i2]R[2][31]..*R[ik][i1]>1 ,则可以进行套汇。
请写一个算法，判断是否可以存在套汇的可能。
【输入格式】
第一行一个整数M代表货币的数量。
以下M行，每行M个整数，空格隔开，代表第i种货币和其他货币的汇率。注意此矩
阵不是对称阵
【输出格式】
0或者-个整数，0代表不存在套汇的可能，大于0代表存在几种套汇的可能。

#include <iostream>
#include <vector>
using namespace std;

const int INF = 1e9;

int main() {
    int M;
    cin >> M;
    vector<vector<double>> rates(M, vector<double>(M));
    for (int i = 0; i < M; ++i) {
        for (int j = 0; j < M; ++j) {
            cin >> rates[i][j];
        }
    }
    for (int k = 0; k < M; ++k) {
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < M; ++j) {
                rates[i][j] = max(rates[i][j], rates[i][k] * rates[k][j]);
            }
        }
    }
    for (int i = 0; i < M; ++i) {
        if (rates[i][i] > 1) {
            cout << "1" << endl;
            return 0;
        }
    }

    // 不存在套汇的可能性
    cout << "0" << endl;

    return 0;
}

2、射击气球

【问题描述】
有一-些球形气球贴在一-堵用XY平面表示的墙面上。墙面.上的气球记录在整数数组
points，其中points[i] = [xstart, xend]表示水平直径在xstart 和xend 之间的气球。.
你不知道气球的确切y坐标。
一支弓箭可以沿着x轴从不同点完全垂直地射出。在坐标x处射出- -支箭,若有一
个气球的直径的开始和结束坐标为xstart， xend， 且满足xstart ≤x≤xend,则该气
球会被引爆。可以射出的弓箭的数量没有限制。弓箭一旦被射出之后，可以无限地前进。
给你一个数组points ，返回引爆所有气球所必须射出的最小弓箭数。
【输入格式】
第一行一个整数M代表气球的的数量。
以下M行，每行两个整数，空格隔开，代表气球的xstart, xend坐标。
【输出格式】
最少需要的弓箭数量

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool compare(const vector<int>& a, const vector<int>& b) {
    return a[1] < b[1];
}

int findMinArrowShots(vector<vector<int>>& points) {
    if (points.empty()) {
        return 0;
    }
    // 按照结束坐标进行排序
    sort(points.begin(), points.end(), compare);
    int arrows = 1;
    int end = points[0][1];
    // 遍历气球，计算最小弓箭数量
    for (int i = 1; i < points.size(); ++i) {
        if (points[i][0] > end) {
            // 如果当前气球的开始坐标大于前一个气球的结束坐标，需要额外的一支箭
            arrows++;
            end = points[i][1];
        }
    }

    return arrows;
}

int main() {
    int M;
    cin >> M;
    vector<vector<int>> points(M, vector<int>(2));
    for (int i = 0; i < M; ++i) {
        cin >> points[i][0] >> points[i][1];
    }
    int result = findMinArrowShots(points);
    cout << result << endl;
    return 0;
}
